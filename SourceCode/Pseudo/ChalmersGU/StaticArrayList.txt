
/* *** ODSATag: StaticArrayList *** */
/* *** ODSATag: StaticArrayListInit *** */
class StaticArrayList(E) implements List(E)
    private internalArray : E[]
    private arraySize     : int

    StaticArrayList(capacity:int)
        internalArray = new array of size capacity
        arraySize = 0
/* *** ODSAendTag: StaticArrayListInit *** */

/* *** ODSATag: StaticArrayListGetSet *** */
    get(i:int) : E
        make sure that 0 <= i < size of internalArray
        return internalArray[i]

    set(i:int, x:E) : E
        make sure that 0 <= i < size of internalArray
        old : E = internalArray[i]
        internalArray[i] = x
        return old
/* *** ODSAendTag: StaticArrayListGetSet *** */

/* *** ODSATag: StaticArrayListAdd *** */
    add(i:int, x:E)
        make sure that arraySize < size of internalArray
        make sure that 0 <= i <= size of internalArray
        arraySize += 1
        for k in arraySize-1, arraySize-2, ..., i+1
            internalArray[k] = internalArray[k-1]
        internalArray[i] = x
/* *** ODSAendTag: StaticArrayListAdd *** */

/* *** ODSATag: StaticArrayListRemove *** */
    remove(i:int) : E
        make sure that size of internalArray > 0
        make sure that 0 <= i < size of internalArray
        x : E = internalArray[i]
        for k in i+1, i+2, ..., arraySize-1
            internalArray[k] = internalArray[k+1]
        arraySize -= 1
        internalArray[arraySize] = null   // For garbage collection
        return x
/* *** ODSAendTag: StaticArrayListRemove *** */

    isEmpty() : boolean
        return arraySize == 0

    size() : int
        return arraySize

    iterator() : Iterator(E)
        return an iterator over the elements in internalArray
/* *** ODSAendTag: StaticArrayList *** */
